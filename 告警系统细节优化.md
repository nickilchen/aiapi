### 2. JavaScript 逻辑优化

#### 加载状态管理
```javascript
// 之前 - 没有显示加载状态
async function loadAlertHistory(page = 1) {
    try {
        const response = await fetch(...);
        const result = await response.json();
        if (result.success) {
            alertHistory = result.data.records;
            renderAlertHistory();
        }
    } catch (error) {
        console.error(error);
    }
}

// 之后 - 正确管理加载状态
async function loadAlertHistory(page = 1) {
    const container = document.getElementById('alertHistoryList');
    
    // 显示加载状态
    container.innerHTML = `
        <div class="loading-state">
            <i class="fas fa-spinner fa-spin"></i>
            <span>加载中...</span>
        </div>
    `;
    
    try {
        const response = await fetch(...);
        const result = await response.json();
        if (result.success) {
            alertHistory = result.data.records;
            renderAlertHistory(); // 会替换加载状态
        } else {
            // 显示错误状态
            container.innerHTML = `
                <div class="empty-state">
                    <i class="fas fa-exclamation-circle"></i>
                    <span>加载失败</span>
                </div>
            `;
        }
    } catch (error) {
        // 显示错误状态
        container.innerHTML = `
            <div class="empty-state">
                <i class="fas fa-exclamation-circle"></i>
                <span>加载失败</span>
            </div>
        `;
    }
}
```
